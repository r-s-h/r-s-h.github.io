for degree in range(1,8):
    fit = numpy.polyfit(range(len(maxes)), maxes, degree)
    poly = numpy.poly1d(fit)
    polyfit = numpy.array([ poly(i) for i in range(len(maxes)*1) ])
    nans = numpy.empty(polyfit.shape)
    nans.fill(numpy.nan)
    nans[:maxes.shape[0]] = numpy.array(maxes)
    stacked = numpy.stack([polyfit, nans])
    df = pd.DataFrame(stacked.T)
    df.columns = ['predicted', 'actual']
    seaborn.lineplot(data = df).set(title="x^%d" % degree)
